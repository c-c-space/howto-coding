type="text/javascript">

let whatButton = document.querySelector('#what button');
let yourName = document.querySelector('#what font');

whatButton.onclick = function() {
  let myName = prompt('What Your Name?');
  if(!myName) {
    setUserName();
  } else {
    localStorage.setItem('name', myName);
    yourName.innerText = `Hello, ${myName}!`;
  }
}

if(!localStorage.getItem('name')) {
  setUserName();
} else {
  let storedName = localStorage.getItem('name');
  yourName.innerText = `Hello, ${storedName}!`;
}
