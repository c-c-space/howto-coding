MouseEvent
ユーザーによるポインティングデバイス (マウス・トラックパッドなど) の操作によってイベントを実行

mousemove
マウスが移動している状態に実行

EventTarget.addEventListener('mousemove', e => {
// JavaScriptを記述
});

EventTarget.onmousemove = function() {
// JavaScriptを記述
} 

document.addEventListener('mousemove', mouseMove);

function mouseMove(xy) {
  mouseXY.innerHTML = `
  Client X/Y: ${xy.clientX} / ${xy.clientY}
  Screen X/Y: ${xy.screenX} / ${xy.screenY}
  `;
}

mousedown
マウスのボタンが押された状態に実行

EventTarget.addEventListener('mousedown', e => {
// JavaScriptを記述
});

EventTarget.onmousedown = function() {
// JavaScriptを記述
}

mouseup

EventTarget.addEventListener('mouseup', e => {
// JavaScriptを記述
});

EventTarget.onmouseup = function() {
// JavaScriptを記述
}

mouseenter
マウスが要素の中に移動したときに実行

EventTarget.addEventListener('mouseenter', e => {
// JavaScriptを記述
});

EventTarget.onmouseenter = function() {
// JavaScriptを記述
}

mouseleave
マウスが要素の外に移動したときに実行

EventTarget.addEventListener('mouseleave', e => {
// JavaScriptを記述
});

EventTarget.onmouseleave = function() {
// JavaScriptを記述
} 

const divs = document.querySelectorAll('div');
for (let i = 0; i < divs.length; i++) {

  divs[i].addEventListener('mouseenter', function(e) {
    e.target.style.backgroundColor = randomRGB();
  });

  divs[i].addEventListener('mouseleave', function(e) {
    e.target.style.backgroundColor = randomRGB();
  });

}
