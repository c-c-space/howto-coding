値を複数のパターンと比較して実行する文を定義する
switch

switch (式) {

case (値)

// 式の値一致した場合に実行する文

break;

default:

// すべての式の値に一致しない場合に実行する文

}

switch 文と if...else 文 の使い方はほとんど同じです。

/* ~~~ */


confirm 関数で呼び出した 質問の回答が もし OK case true: なら I'm OK! とアラートを表示し、 回答が それ以外 default なら I'm CANCEL! とアラートを表示する 

 type="text/javascript">

'use strict'

function switchAnsewer() {
  let okCancel = confirm('Are you OK or CANCEL ?')

  switch (okCancel) {
    case true:
    alert( "I'm OK!" );
    break;

    default:
    alert( "I'm CANCEL!" );
  }
}

/* ~~~ */


prompt 関数で呼び出した 入力フィールドが もし キャンセル case null された場合は、 Top Secret! とアラート を表示し、 0以下 (未記入) case '0': case '': だった場合は、 In Utero とアラートを表示し、 それ以外 default の場合は、 その値を含むアラートを表示する 

 type="text/javascript">

'use strict'

function switchAge() {
  let howold = prompt('How old are you?', 0)

  switch (howold) {
    case null:
    alert( "Top Secret!" );
    break;

    case '0':
    case '':
    alert( 'In Utero' );
    break;

    default:
    alert( `I'm ${howold} years old!` );
  }
}
