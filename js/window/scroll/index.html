Scroll & WheelEvent

スクロールイベント
文書のビューまたは要素がスクロールされたときに実行する

EventTarget.addEventListener('scroll', event => {
// JavaScriptを記述
});

EventTarget.onscroll = function() {
// JavaScriptを記述
} 

document.body.onscroll = (event) => {
  // スクロール状態に実行する文

  setTimeout(() => {
    // スクロールが止まった状態に実行する文
  }, 0);
};

window.scrollTo
ウィンドウを文書内の特定の位置までスクロールする

ページトップにスクロール

window.scrollTo ({
top: 0,
behavior: "smooth"
});

ページ最下部にスクロール

let elm = document.documentElement;
let bottom = delm.scrollHeight - elm.clientHeight;

window.scrollTo ({
top: bottom
behavior: "smooth"
});

window.scrollBy
ウィンドウ内の文書を指定された量だけスクロールさせる

1 ページ分下スクロール

1 ページ分上スクロール

window.scrollBy ({
top: window.innerHeight,
behavior: "smooth"
});

WheelEvent
ユーザーがマウスホイールやそれに似た機器を動かしたときに実行する

EventTarget.addEventListener('wheel', event => {
// JavaScriptを記述
});

EventTarget.onwheel = function() {
// JavaScriptを記述
} 

let scale = 1;

wheel.addEventListener('wheel', event => {
  event.preventDefault();

  scale += event.deltaY * -0.01;

  // Restrict scale
  scale = Math.min(Math.max(.5, scale), 1.5);

  // Apply scale transform
  wheel.style.transform = `scale(${scale})`;
});

deltaX
マウスホイールのX軸方向（横方向）のスクロール量

deltaY
マウスホイールのY軸方向（縦方向）のスクロール量

deltaZ
マウスホイールのZ軸方向のスクロール量

deltaMode
スクロール量の単位
