Math.floor()
() に 指定した数値以下の最大の整数を返す
Math.random()
乱数を生成する範囲を設定する場合は、() * の後に 数値を指定 

function random(min, max) {
  const num = Math.floor(Math.random() * (max - min + 1)) + min;
  return num;
}

function randomRGB() {

オブジェクトの配列順をランダムにシャッフルする
shuffle(Object.entries(オブジェクト名)) 

function shuffle(arrays) {
  const array = arrays.slice();
  for (let i = array.length - 1; i >= 0; i--) {
    const shuffleArr = Math.floor(Math.random() * (i + 1));
    [array[i], array[shuffleArr]] = [array[shuffleArr], array[i]];
  }
  return array;
};

let オブジェクト名 = {

  '1' : 'A',
  '2' : 'B',
  '3' : 'C',

}


function shuffleDl() {
  const allShuffle = shuffle(Object.entries(オブジェクト名));
  const shuffleList = document.querySelector('dl#shuffleList');
  shuffleList.innerText = '';

  allShuffle.forEach(eachArr => {
    const shuffleDt = document.createElement ('dt');
    shuffleDt.innerText = Object.values(eachArr[0]);
    shuffleList.appendChild(shuffleDt);

    const shuffleDd = document.createElement ('dd');
    shuffleDd.innerText = Object.values(eachArr[1]);
    shuffleList.appendChild(shuffleDd);
  });
}
