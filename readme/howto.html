.readme セクション要素とアンカーリンク

#one">ONE

#two">TWO

#three">THREE

#four">FOUR

class="readme">

id="one" class="active">

id="two" class="is-hide">

id="three" class="is-hide">

id="four" class="is-hide">

src="script.js"> 
"use strict"

const anchorLinks = document.querySelectorAll('a[href^="#"]');
const anchorLinksArr = Array.prototype.slice.call(anchorLinks);

anchorLinks.forEach(link => {
  link.addEventListener('click', event => {
    event.preventDefault();
    const readmeSectionAll = document.querySelectorAll(".readme section");
    for (const readmeSection of readmeSectionAll) {
      readmeSection.classList.remove("active");
      readmeSection.classList.add("is-hide");
    }
    const targetId = link.hash;
    const targetElement = document.querySelector(targetId);
    targetElement.classList.replace("is-hide", "active");
  });
});

アンカーリンク a[href^="#"] がクリックされたら .readme section 要素のクラスリストに .is-hide を追加 .active を取り除き、 アンカーリンク先の要素のクラスリストは .is-hide から .active に置き換える
